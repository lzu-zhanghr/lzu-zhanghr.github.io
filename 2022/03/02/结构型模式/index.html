适配器模式

  • 内容：将一个类的接口转换成客户希望的另一个接口，使由于接口不兼容而无法共同工作的类可以协作
  • 实现方式：
    • 类适配器：多继承
    • 对象适配器：组合
from abc import ABCMeta, abstractmethod

class Payment(metaclass=ABCMeta):
    @abstractmethod
    def pay(self, money):
        pass

class Alipay(Payment):
    def pay(self, money):
        print(f"支付宝支付{money: d}元")

class WechatPay(Payment):
    def pay(self, money):
        print(f"微信支付{money: d}元")

class BankPay:
    def cost(self, money):
        print(f"银联支付{money: d}元")

# 使用多继承实现适配器，使接口保持了一致

class InheritBankPay(Payment, BankPay):
    def pay(self, money):
        self.cost(money)

p1 = InheritBankPay()
p1.pay(200)

# 使用组合类实现适配器，使接口保持了一致

class CompositeBankPay(Payment):
    def __init__(self, payment):
        self.payment = payment

    def pay(self, money):
        self.payment.cost(money)

p2 = CompositeBankPay(BankPay())
p2.pay(100)

桥模式

  • 内容：将一个事物的两个维度分离，使其可以独立的变化, 可以很便利的在不同维度添加对象
  • 优点：
    • 抽象和实现相互分离
    • 优秀的扩展能力
from abc import ABCMeta, abstractmethod

class Shape(metaclass=ABCMeat):
    def __init__(self, color):
        self.color = color

    @abstractmethod
    def draw(self):
        pass

class Color:
    @abstractmethod
    def paint(self, shape):
        pass

class Rectangle(Shape):
    name = "rectangle"
    def draw(self):
        self.color.paint(self)

class Triangle(Shape):
    name = "Triangle"
    def draw(self):
        self.color.paint(self)

class Red(Color):
    def paint(self, shape):
        print(f"red {shape: s}")

class Green(Color):
    def paint(self, shape):
        print(f"green {shape: s}")

shape1 = Rectangle(Green())
shape1.draw()

shape2 = Triangle(Red())
shape2.draw()
