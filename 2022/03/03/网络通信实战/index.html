网络通信实战

网络通信实战

socket

简介

  • socket 并非 OSI, 或者 TCP/IP 参考模型中的某一个协议， 而是程序员在实际应用编程中总结出的一个规范化的通信编程接口，可以实现不同程序间相互通信，包括本地进程间通信和网络进程间通信；实现网络通信时，其位置看做位于 TCP/IP 模型中应用层和传输层之间；

简单实现

import socket

# server

sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
sock.bind(("0.0.0.0", 12000))
sock.listen(5)

while True:
    connection, address = sock.accept()

    client_data = connection.recv(1024)
    print(client_data.decode("utf-8"))

    connection.sendall("nice to meet you, too.".encode("utf-8"))
    connection.close()

# client

client = socket.socket()

client.connect(("0.0.0.0", 12000))
client.sendall("nice to meet you".encode("utf-8"))

reply = client.recv(1024)
print(reply.decode("utf-8"))

client.close()

import socket
import select

# server

server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
server.setblocking(False)
server.bind(("0.0.0.0", 12000))
server.listen(5)

inputs = [server, ]

while True:
    # rlist: wait until ready for reading  
    # wlist: wait until ready for writing  
    # xlist: wait for an “exceptional condition
    r, w, e = select.select(inputs, [], [], 0.05)
    for sock in r:
        if sock == server:
            conn, addr = sock.accept()
            print("有新连接出现")
            inputs.append(conn)
        else:
            data = sock.recv(1024)
            if data:
                print("收到消息："， data)
            else:
                print("关闭连接")
                inputs.remove(sock)

import socket

# server

sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
sock.bind(("0.0.0.0", 12000))
sock.listen(5)

while True:
    conn, addr = sock.accpet()
    data = conn.recv(1024)
    total_file_size = int(data.decode("utf-8"))

    with open('xxx.png', 'ab') as f:
        recv_size = 0  
        while True:
            data = conn.recv(1024)
            f.write(data)
            f.flush()

            recv_size  += len(data)
            if recv_size == total_file_size:
                break

    conn.close()

import struct

# server

sock = socket.socket()  
sock.bind(("0.0.0.0", 12000))
sock.listen(5)

while True:
    conn, addr = sock.accept()
    header = conn.recv(4)
    data_len = struct.unpack("i", header)[0]
