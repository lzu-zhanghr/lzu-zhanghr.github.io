异常处理

try, except, else

try:
    f = open("test_file", "w")
    f.write("test error case")
except IOError:
    print("file not found")
else:
    print("successfully write file")
    f.close()

try, finally

try:
    f = open("test_file", "w")
    try:
        f.write("test error case")
    finally:
        f.close()
except IOError:
    print("file not found")

自定义异常与抛出

import Exception

class GenderException(Exception):
    def __init__(self, *arg):
        super().__init__()
        self.args = arg

gender ="male-female"
if gender != "male" and gender !="female":
    raise GenderException("only male and female")
