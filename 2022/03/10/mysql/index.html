必备 SQL 指令

use database_name;
create table depart(
    id int not null auto_increment primary key,
    titlr varchar(20) not null
)default charset=utf8;
create table info(
    id int not null auto_increment primary key,
    name varchar(20) not null,
    email varchar(30) not null,
    age int,
    depart_id int 
)default charset=utf8;


-- 条件查询
select * from info where age > 30;
select * from info where age != 30;
select * from info where age between 2 and 4;

select * from info where name = "ccc" and age = 25;
select * from info where name = "ccc" or age = 25;
select * from info where name = (1, 2, 3);

-- 子查询
select * from info where id in (select id from depart);
select * from info where exists (select * from depart where id = 5);
select * from info where not exists (select * from depart where id = 5);
select from (select * from info where id > 5) as T where T.age > 10;


-- 通配符
-- % 占位 n 个字符, _ 占位一个 字符
select * from info where name like "%沛%";
select * from info where email like "__peiqi@live.com";


-- 映射, 效率很低
select 
    id,
    name,
    ( select title from depart where depart.id = info.id) as x,
    -- ( select title from depart where depart.id = info.depart_id) as x,
from info;


-- 排序
select * from info order by age desc;
select * from info order by age asc;
select * from info order by age asc, id desc;


-- 部分数据操作
select * from info limit 5;
select * from info order by id desc limit 3;
select * from info limit offset 2; -- offset 设置开始取的位置


-- 分组
select age, max(id), min(id), count(name) from info group by age;
select depart_id, count(id) from info group by depart_id;
select depart_id, count(id) from info group by depart_id having count(id) > 2; -- 这里分组之后不能用 where， 用having


-- 左右连接
-- 多表联合查询
select * from info left outer join depart on info.depart_id = depart.id;
-- 主表，从表的确定:要展示谁的数据就以谁为主表，一般最好用左外连接
select * from info left join depart on info.depart_id = depart.id -- 简写

-- 内连接：会忽略两张表没有联系的属性
select * from info inner join depart on info.depart_id = depart.id;


-- 联合 union: 列数量相同就可以连接
select id, title from depart 
union 
select id, name from info; -- 会自动去重

select id, title from depart 
union all
select id, name from info; -- 不会自动去重

表关系

-- 外键约束：保证一个列的值必须是其他表中特定列已经存在的值。

-- 创建时约束：
create table depart(
    id int not null auto_increment primary key,
    title varchar(16) not null
)default charset=utf8;

create table info(
    id int not null auto_increment primary key,
    name varchar(16) not null,
    email varchar(32) not null,
    age int,
    depart_id int not null,
    constraint fk_info_depart foreign key (depart_id) references depart(id)
)default charset=utf8;

-- 创建后添加：
alter table info add constraint fk_into_depart foreign key info(depart_id) references depart(id);
-- 删除外键
alter table info drop foreign key fk_info_depart;

数据库索引

-- 创建索引，加速查找，加入索引之后，数据基于 B+Tree 数据结构存储，加速了查询，减慢了增加删除改动操作速度。
-- 非聚簇索引(mysiam 引擎)
create table name(
    id int not null auto_increment primary key,
    name varchar(32) not null,
    age int,
)engine=mysiam default charset=utf8;
-- 聚簇索引(innodb 引擎)
create table name(
    id int not null auto_increment primary key,
    name varchar(32) not null,
    age int
)engine=innodb default charset=utf8; -- 默认引擎

-- 实际使用
create table name(
    id int not null auto_increment primary key, --主键索引
    unique name varchar(32) not null, --唯一索引
    index age int --普通索引
)default charset=utf8;

-- 命中索引注意事项，
    -- 1.字符类型保持一致
    -- 2.不能使用不等于运算符
    -- 3.like 通配符不能命中

-- 判断是否命中索引
-- explain + sql 语句
explain select * from big where name=123;

安全锁

  • update,insert,delete 内部会加锁，select 不会加锁
    -- select需要加锁的时候
-- 排他锁
begin; 
    select * from tablename where name=key_name for update;
commit;
-- 共享锁:允许加锁之后查询，但不允许修改
begin; 
    select * from tablename where name=key_name lock in share mode;
commit;
