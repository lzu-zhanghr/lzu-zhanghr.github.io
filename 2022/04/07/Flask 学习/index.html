class BaseConverter:

    regex = '[^/]+'
    weight = 100

        self.map = map

    def to_python(self, value):
        return value

    def to_url(self, value):
        return url_quote(value, charset=self.map.charset)

from flask import Flask, request

请求钩子

  • 应用场景：客户端与服务端交互过程中，有些准备工作或者扫尾工作需要处理，包括：
    • 请求开始时：建立的数据库连接
    • 请求开始时：根据需求进行权限校验
    • 请求结束时：指定数据交互格式
  • 功能实现：为了 避免每个视图函数编写重复代码，Flask 提供了装饰器实现的钩子，包括以下四种：
    • before_first_request: 处理第一个请求前执行
    • before_request: 每次请求前执行，该装饰器函数一旦 return, 视图函数不再执行
    • after_request: 如果没有抛出错误，在每次请求后执行
      • 接受一个参数：视图函数做出响应，在返回之前对响应值做最后一步处理而后返回
    • teardown_request: 每次请求之后执行
      • 接受一个参数：用来接受错误信息
from flask import Flask

app = Flask(__name__)

# 初始化操作, io文件读写, 数据库连接
@app.before_first_request
def before_first_request():
    print('before_first_request')

# 参数校验，访问量统计
@app.before_request
def before_request():
    print('before_request')

# 适合对返回值格式做统一处理
@app.after_request
def after_request(resp):
    print('after_request')
    return resp

# 请求销毁之后执行，适合做异常信息统计
@app.teardown_request
def after_request(error):
    print('teardown_request')

@app.route('/')
def hello_world():
    print('hello flask')
    return 'hello flask'

if __name__ == '__main__':
    app.run()

状态保持

  • 相关概念
    • 无状态：指一次用户请求时，浏览器、服务器不知道该用户之前做过什么，每次请求都是一次新的请求
    • Cookie: 在客户端存储状态信息
      • 设置 cookie: response.set_cookie(key, value, max_age)
      • 获取 cookie: request.cookie.get(‘key’)
    • Session: 在服务端存储状态信息，主要存储一些不能存储在浏览器的敏感信息，包括密码，信用卡账号；session 必须使用 cookie 存储， session id 必须加密
      • 设置 session: session[key] = value
      • 获取 session: value = session.get(key)
from flask import Flask, make_response

app = Flask(__name__)

@app.route('/set_cookie')
def set_cookie():
    response = make_response('set cookie')
    response.set_cookie('computer', 'lenovo', 10)
    return response

@app.route('/get_cookie')
def get_cookie():
    name = request.cookie.get('computer')

    return f'name is {name: s}.'

上下文

  • 请求上下文: 在 flask 中，可以直接在视图函数中使用 request 对象获取相关数据，保存了当前请求相关数据；请求上下文对象有：request, session

    • request: 封装了 http 请求内容，针对的是 http 请求
    • session: 用于记录请求会话中的信息，针对的是用户信息

  • 应用上下文：应用上下文用于存储应用程序中的变量，可以通过 current_app.name 获取当前 app 名称，存储的变量包括：

    • 应用的启动脚本是哪个，启动时指定了什么参数
    • 加载了哪些配置文件，导入了哪些配置
from flask import Flask, make_response

app = Flask(__name__)

@app.route('/')
def hello_world():
    print(app.config.get('DEBUG'))

if __name__ == '__main__':
    app.run(debug=True)

CSRF(cross site request forgery)

  • 指攻击者盗用了身份信息，而后发送恶意请求
from flask_wtf.csrf import generate_csrf

@app.after_request
def set_csrf_cookie(response):
    response.set_cookie('csrf-token', generate_csrf())

    return response

ORM(object relation mapping)

  • 优点：

    • 对数据库的操作都转化成对类，属性和方法的操作。
    • 不用编写各种数据库的 SQL 语句
    • 不再关注使用的是什么数据库类型

  • 缺点：

    • 存在性能损失

  • 增删改

from flask import Flask
from flask_sqlalchemy import SQLAlchemy

app = Flask(__name__)

app.config['SQLALCHEMY_DATABASE_URI'] = 'mysql+pymysql'

db = SQLAlchemy(app)

class Student(db.Model):
    __tablename__ == 'students'
    id = db.Column(db.Integer, primary_key=True)
    name = db.Column(db.String(32))
    show = db.Column(db.Boolean, default=True)
    teacher_id = db.Column(db.Integer, db.ForeignKey(Teacher.id))
    courses = db.relationship('Course', secondary='courses', lazy='subquery', backref=db.backref('students', lazy=True))


class Teacher(db.Model):
    __tablename__ == 'teachers'
    id = db.Column(db.Integer, primary_key=True)
    name = db.Column(db.String(32))
    show = db.Column(db.Boolean, default=True)
    students = db.relationship('Student', backref='teacher', lazy=True)

class Course(db.Model):
    __tablename__ = 'courses'
    id = db.Column(db.Integer, primary_key=True)
    name = db.Column(db.String(32))

courses = db.Table('courses',
    db.Column('course_id'), db.Integer, db.ForeignKey('course.id'), primary_key=True,
    db.Column('student_id'), db.Integer, db.ForeignKey('student.id'), primary_key=True,
    )

@app.route('/')
def hello_world():
    teacher = Teacher(name='john')
    db.session.add(teacher)
    db.session.commit()
    
    student = Student(name='alex', teacher_id=teacher.id)
    db.session.add(student)
    db.session.commit()

    t1 = Teacher(name='smith')
    t2 = Teacher(name='tom')
    db.session.add_all([t1, t2])
    db.session.commit()
    db.session.delete(t1)
    db.session.delete(t2)
    db.session.commit()

    db.session.rool_back()
    db.session.remove()
    return 'hello world'

if __name__ == '__main__':
    db.create_all()
    app.run(debug=True)
